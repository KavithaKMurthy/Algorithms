using System;

public class HelloWorld
{
    public static void Main(string[] args)
    {
        int[] nums = new int[] { -2, 1, -3, 4, -1, 2, 1, -5, 4 };
        int maxSum = MaximumSubArraySum(nums);
        Console.WriteLine ("Maximum SubArray Sum is {0}", maxSum);
    }
    
    private static int MaximumSubArraySum(int[] nums)
    {
        int currSum = 0;
        int maxSum = 0;
        
        for(int i = 0; i < nums.Length ; i++)
        {
            if(currSum < 0)
           {
               currSum = 0;
           }
           
           currSum += nums[i];
           
           if(currSum > maxSum)
           {
               maxSum = currSum;
           }
        }
        return  maxSum;
    }
}
